#### 2.2 添加钉钉登录接口

在 `server-minimal.js` 中添加：

```javascript
const axios = require('axios');
const crypto = require('crypto');

// 钉钉配置
const DINGTALK_CONFIG = {
    appKey: '你的AppKey',
    appSecret: '你的AppSecret',
    agentId: '你的AgentId'
};

// 获取钉钉访问令牌
async function getDingTalkToken() {
    const res = await axios.get('https://oapi.dingtalk.com/gettoken', {
        params: {
            appkey: DINGTALK_CONFIG.appKey,
            appsecret: DINGTALK_CONFIG.appSecret
        }
    });
    
    if (res.data.errcode === 0) {
        return res.data.access_token;
    } else {
        throw new Error('获取钉钉令牌失败: ' + res.data.errmsg);
    }
}

// 钉钉登录接口
app.post('/api/auth/dingtalk-login', async (req, res) => {
    const { code } = req.body;
    
    if (!code) {
        return res.json({ success: false, message: '缺少授权码' });
    }
    
    try {
        const accessToken = await getDingTalkToken();
        
        // 获取用户基本信息
        const userRes = await axios.post(`https://oapi.dingtalk.com/topapi/v2/user/getuserinfo?access_token=${accessToken}`, {
            code: code
        });
        
        if (userRes.data.errcode !== 0) {
            return res.json({ success: false, message: '获取用户信息失败: ' + userRes.data.errmsg });
        }
        
        const userId = userRes.data.result.userid;
        
        // 获取用户详情
        const detailRes = await axios.post(`https://oapi.dingtalk.com/topapi/v2/user/get?access_token=${accessToken}`, {
            userid: userId
        });
        
        if (detailRes.data.errcode !== 0) {
            return res.json({ success: false, message: '获取用户详情失败' });
        }
        
        const userInfo = detailRes.data.result;
        
        // 查找或创建本地用户
        db.get('SELECT * FROM users WHERE dingtalk_id = ? OR username = ?', [userId, userInfo.mobile], (err, user) => {
            if (err) {
                return res.json({ success: false, message: '数据库查询失败' });
            }
            
            if (user) {
                // 用户已存在，更新钉钉ID（如果没有）
                if (!user.dingtalk_id) {
                    db.run('UPDATE users SET dingtalk_id = ? WHERE id = ?', [userId, user.id]);
                }
                
                // 生成JWT令牌
                const token = jwt.sign(
                    { userId: user.id, username: user.username, role: user.role },
                    JWT_SECRET,
                    { expiresIn: '24h' }
                );
                
                res.json({
                    success: true,
                    token,
                    user: {
                        id: user.id,
                        username: user.username,
                        realName: user.real_name,
                        role: user.role
                    }
                });
            } else {
                // 创建新用户
                const newUser = {
                    username: userInfo.mobile || userId,
                    password: '',  // 钉钉登录无需密码
                    real_name: userInfo.name,
                    dingtalk_id: userId,
                    department: userInfo.dept_name || '',
                    mobile: userInfo.mobile || '',
                    role: 'user'
                };
                
                db.run(`INSERT INTO users (username, password, real_name, dingtalk_id, department, mobile, role) 
                        VALUES (?, ?, ?, ?, ?, ?, ?)`,
                    [newUser.username, newUser.password, newUser.real_name, newUser.dingtalk_id, 
                     newUser.department, newUser.mobile, newUser.role],
                    function(err) {
                        if (err) {
                            return res.json({ success: false, message: '创建用户失败: ' + err.message });
                        }
                        
                        const token = jwt.sign(
                            { userId: this.lastID, username: newUser.username, role: newUser.role },
                            JWT_SECRET,
                            { expiresIn: '24h' }
                        );
                        
                        res.json({
                            success: true,
                            token,
                            user: {
                                id: this.lastID,
                                username: newUser.username,
                                realName: newUser.real_name,
                                role: newUser.role
                            }
                        });
                    }
                );
            }
        });
    } catch (error) {
        console.error('钉钉登录失败:', error);
        res.json({ success: false, message: '登录失败: ' + error.message });
    }
});
```

#### 2.4 创建钉钉回调页面

创建 `frontend/dingtalk-callback.html`：

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>钉钉登录中...</title>
    <script src="https://unpkg.com/axios/dist/axios.min.js"></script>
</head>
<body>
    <div style="text-align: center; margin-top: 100px;">
        <h2>正在登录...</h2>
        <p>请稍候</p>
    </div>
    
    <script>
        // 获取URL中的code
        const urlParams = new URLSearchParams(window.location.search);
        const code = urlParams.get('code');
        const state = urlParams.get('state');
        
        // 验证state（防止CSRF攻击）
        const savedState = localStorage.getItem('dingtalk_state');
        if (state !== savedState) {
            alert('登录失败：状态验证错误');
            window.location.href = '/login.html';
        }
        
        // 发送code到后端
        axios.post('/api/auth/dingtalk-login', { code })
            .then(res => {
                if (res.data.success) {
                    // 保存token
                    localStorage.setItem('token', res.data.token);
                    localStorage.setItem('user', JSON.stringify(res.data.user));
                    
                    // 跳转到首页
                    window.location.href = '/index.html';
                } else {
                    alert('登录失败：' + res.data.message);
                    window.location.href = '/login.html';
                }
            })
            .catch(err => {
                alert('登录失败：' + err.message);
                window.location.href = '/login.html';
            });
    </script>
</body>
</html>
```

## 📊 方案3：完整生产部署（企业级）

### 架构说明

```
[用户] 
   ↓
[负载均衡 SLB]
   ↓
[Nginx 集群] ← [SSL证书]
   ↓
[Node.js 应用集群] ← [PM2]
   ↓
[MySQL 主从] + [Redis 缓存]
```

### 关键步骤

#### 1. 数据库优化

将SQLite迁移到MySQL，并配置主从复制：

```bash
# 主库配置（/etc/mysql/mysql.conf.d/mysqld.cnf）
[mysqld]
server-id = 1
log_bin = /var/log/mysql/mysql-bin.log
binlog_do_db = talent_assessment

# 从库配置
[mysqld]
server-id = 2
relay-log = /var/log/mysql/mysql-relay-bin.log
read-only = 1
```

#### 2. 配置Redis缓存

```bash
sudo apt-get install -y redis-server
sudo systemctl enable redis
sudo systemctl start redis
```

在代码中添加Redis缓存：

```javascript
const redis = require('redis');
const client = redis.createClient();

// 缓存用户信息
function cacheUser(userId, userInfo) {
    client.setex(`user:${userId}`, 3600, JSON.stringify(userInfo));
}

// 获取缓存
function getCachedUser(userId, callback) {
    client.get(`user:${userId}`, (err, data) => {
        if (data) {
            callback(null, JSON.parse(data));
        } else {
            callback(err, null);
        }
    });
}
```

#### 3. 配置PM2集群

创建 `ecosystem.config.js`：

```javascript
module.exports = {
  apps: [{
    name: 'talent-assessment',
    script: 'server-minimal.js',
    instances: 'max',  // 使用所有CPU核心
    exec_mode: 'cluster',
    env: {
      NODE_ENV: 'production',
      PORT: 3000
    },
    error_file: '/var/log/talent-assessment/error.log',
    out_file: '/var/log/talent-assessment/out.log',
    log_date_format: 'YYYY-MM-DD HH:mm:ss'
  }]
};
```

启动：

```bash
pm2 start ecosystem.config.js
pm2 save
```

#### 4. 配置监控

```bash
# 安装PM2监控模块
pm2 install pm2-logrotate  # 日志轮转
pm2 set pm2-logrotate:max_size 10M
pm2 set pm2-logrotate:retain 7

# 实时监控
pm2 monit

# 查看日志
pm2 logs talent-assessment
```

#### 5. 配置自动备份

创建备份脚本 `/opt/backup-talent.sh`：

```bash
#!/bin/bash

DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR=/var/backups/talent-assessment

# 创建备份目录
mkdir -p $BACKUP_DIR

# 备份MySQL
mysqldump -u talent_user -p'Talent@2026!' talent_assessment > $BACKUP_DIR/db_$DATE.sql

# 压缩备份
gzip $BACKUP_DIR/db_$DATE.sql

# 备份上传文件
tar -czf $BACKUP_DIR/uploads_$DATE.tar.gz /var/www/talent-assessment/uploads

# 删除7天前的备份
find $BACKUP_DIR -name "*.sql.gz" -mtime +7 -delete
find $BACKUP_DIR -name "*.tar.gz" -mtime +7 -delete

echo "备份完成: $DATE"
```

设置定时任务：

```bash
chmod +x /opt/backup-talent.sh

# 编辑crontab
crontab -e

# 添加以下行（每天凌晨2点备份）
0 2 * * * /opt/backup-talent.sh >> /var/log/talent-assessment/backup.log 2>&1
```

## 👥 用户管理方案

### 方案A：开放注册（适合外部使用）

在登录页面添加"注册"按钮，允许用户自己注册。

**实现步骤**：

2. 填写用户信息（姓名、部门、邮箱）
3. 上传Excel文件
4. 系统自动创建账号并发送邮件

同事点击钉钉登录，系统自动创建账号，无需记忆密码。

**优点**：无缝集成，用户体验好
**缺点**：需要钉钉管理员权限

    db.run(`INSERT INTO login_logs (user_id, login_time, ip_address, user_agent) 
            VALUES (?, datetime('now'), ?, ?)`,
            [user.id, req.ip, req.get('User-Agent')]);
});

// 统计在线用户
app.get('/api/stats/online', auth, (req, res) => {
    db.get(`SELECT COUNT(DISTINCT user_id) as count 
            FROM login_logs 
            WHERE login_time > datetime('now', '-1 hour')`,
            (err, row) => {
                res.json({ success: true, onlineCount: row.count });
            });
});

// 统计测评完成情况
app.get('/api/stats/assessments', auth, (req, res) => {
    db.all(`SELECT t.name, COUNT(*) as count 
            FROM assessment_results r
            JOIN assessment_tools t ON r.tool_id = t.id
            GROUP BY r.tool_id`,
            (err, rows) => {
                res.json({ success: true, stats: rows });
            });
});

## 💡 常见问题解答

### Q1: 同事如何注册使用？

**答**：有三种方式：
1. 开放注册：同事自己注册账号
2. 批量导入：管理员导入，系统发邮件通知
3. 钉钉登录：点击钉钉登录，自动创建账号（推荐）

### Q2: 如何控制访问权限？

**答**：在数据库中配置角色：
- 管理员：全部权限
- HR：查看所有报告
- 部门经理：查看本部门报告
- 普通员工：只能查看自己的报告

### Q3: 如何保证数据安全？

**答**：
1. 配置HTTPS加密传输
2. 定期备份数据库
3. 限制数据库端口访问
4. 使用强密码和JWT认证

### Q4: 系统能支持多少人同时使用？

**答**：
- 2核4G服务器：支持50人同时在线
- 4核8G服务器：支持200人同时在线
- 8核16G服务器：支持500人同时在线

### Q5: 如何获取技术支持？

**答**：
1. 查看系统日志：`pm2 logs talent-assessment`
2. 查看浏览器控制台（F12）
3. 联系开发者（如果您是自己部署的）
